    for index, row in df.iterrows():
        first_name = row['First Name (Given Name)']
        last_name = row['Last Name (Family Name)']
        
        f.write(f"EMPLOYEE: {first_name} {last_name}\n")

        citizenship_status = row['Citizenship/Immigration Status']
        if "citizen of the United States" in citizenship_status:

            i9_status = "3. A lawful permanent resident"
        elif "noncitizen authorized" in citizenship_status:
            i9_status = "4. A noncitizen authorized to work"
        else:

        f.write(f"Address: {row['Address (Street Number and Name)']}\n")
        f.write(f"Apt Number: {row['Apt. Number (if any)']}\n")

        f.write(f"State: {row['State']}\n")
        f.write(f"ZIP Code: {row['ZIP Code']}\n")

        f.write(f"Telephone: {row['Employee\\'s Telephone Number']}\n")

        # Additional fields for noncitizens

            f.write(f"USCIS A-Number: {row['USCIS A-Number']}\n")
